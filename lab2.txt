BEGIN
 
DECLARE CONSTANT POUNDS_PER_INCH, BASE_HEIGHT, BASE_WEIGHT

// Input Module

CALL input_module TO GET POUNDS_PER_INCH, BASE_HEIGHT, BASE_WEIGHT, total_height

// Calculate Module

CALL calculate_module TO GET ideal_weight

// Output Module

CALL output_module TO DISPLAY height, ideal_weight

END

MODULE input_module

// Constants

SET POUNDS_PER_INCH = 5
SET BASE_HEIGHT = 60
BASE_WEIGHT = 110

// Input

OUTPUT "Enter the height in full feet: "
INPUT height_feet

OUTPUT "Enter the additional inches in height: "
INPUT height_inches

SET total_height = height_feet * 12 + height_inches

END MODULE

MODULE calculate_module

//Calculate Height

SET calculated_height = total_height - BASE_HEIGHT

// Calculate Weight

SET ideal_weight = calculated_height * POUNDS_PER_INCH + BASE_WEIGHT

END MODULE

MODULE output_module

// Output Height and Ideal Weight

SET output_height = total_height / 12
OUTPUT "User's height in feet: ", output_height
OUTPUT "Calculated ideal body weight: ", ideal_weight, "pounds"

END MODULE
