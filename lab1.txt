Begin

OUTPUT "How many students are registered in the 2-year program"
Input year2Students

OUTPUT "How many students are registered in the 3-year program"
Input year3Students

SET totalStudents = year2Students + year3Students

OUTPUT "The total number of students in the class:" + totalStudents

SET percentage2Year = year2Students / totalStudents * 100
SET percentage3Year = year3Students / totalStudents * 100

OUTPUT "The percentage of 2-year program students is:" + percentage2Year
OUTPUT "The percentage of 3-year program students is:" + percentage3Year

END
